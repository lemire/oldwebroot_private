%------------------------------------------------------------------------
% Macros to include figures (Encapsulated PostScript only)
%------------------------------------------------------------------------

%------------------------------------------------------------------------
% \psone{}{} allows you to insert one eps figure
%------------------------------------------------------------------------
% #1 filename of the figure (must be in current directory)
% #2 horizontal size for the figure in mm (0 to keep natural size)
%------------------------------------------------------------------------

\def\psone#1#2{\centerline{\epsfxsize=#2mm \epsfbox{#1}}}

%------------------------------------------------------------------------
% \pstwo{}{}{}{} allows you to insert side by side two eps figures
%------------------------------------------------------------------------
% #1 filename of the 1st figure (must be in current directory)
% #2 horizontal size for the 1st figure in mm (0 to keep natural size)
% #3 filename of the 2nd figure (must be in current directory)
% #4 horizontal size for the 2nd figure in mm (0 to keep natural size)
%------------------------------------------------------------------------

\def\pstwo#1#2#3#4{\centerline{%
\epsfxsize=#2mm \epsfbox{#1}\qquad
\epsfxsize=#4mm \epsfbox{#3}
}}%

%------------------------------------------------------------------------
% \psthree{}{}{}{}{}{} allows you to insert side by side 3 eps figures
%------------------------------------------------------------------------
% #1 filename of the 1st figure (must be in current directory)
% #2 horizontal size for the 1st figure in mm (0 to keep natural size)
% #3 filename of the 2nd figure (must be in current directory)
% #4 horizontal size for the 2nd figure in mm (0 to keep natural size)
% #5 filename of the 3rd figure (must be in current directory)
% #6 horizontal size for the 3rd figure in mm (0 to keep natural size)
%------------------------------------------------------------------------

\def\psthree#1#2#3#4#5#6{\centerline{%
\epsfxsize=#2mm \epsfbox{#1}\qquad
\epsfxsize=#4mm \epsfbox{#3}\qquad
\epsfxsize=#6mm \epsfbox{#5}
}}%

%------------------------------------------------------------------------
% Macros to insert legends under figures
%------------------------------------------------------------------------

%------------------------------------------------------------------------
% \figlegend{}{}{} allows you to put some text under a figure
%------------------------------------------------------------------------
% #1 : the number you want to associate to the figure
% #2 : the text of your legend
% #3 : the (possibly <0) vertical offset ( in mm) between fig. and text 
%------------------------------------------------------------------------

\def\figlegend#1#2#3{%
\null\vskip 1pt\vskip #3mm
{\centerline{{\bf Fig.~#1.} #2.}}%
}

%------------------------------------------------------------------------
% use \figlonglegend{}{}{} when the legend doesn't fit in one line
%------------------------------------------------------------------------
% #1 : the number you want to associate to the figure
% #2 : the text of your legend
% #3 : the (possibly <0) vertical offset (in mm) between fig. and text 
%------------------------------------------------------------------------

\def\figlonglegend#1#2#3{%
\null\vskip 1pt\vskip #3mm
{{\narrower \noindent{\bf Fig.~#1.} #2.\smallskip}}%
}
